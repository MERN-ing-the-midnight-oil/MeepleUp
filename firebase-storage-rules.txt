rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    
    // Profile pictures - using regex pattern matching for better reliability
    match /profile-pictures/{fileName} {
      // Helper function to extract userId from filename
      function getUserIdFromFilename() {
        return fileName.split('_')[0];
      }
      
      // Allow delete if authenticated and filename matches user's UID pattern
      // (request.resource doesn't exist for delete operations)
      allow delete: if request.auth != null && 
                       fileName.matches(request.auth.uid + '_[0-9]+\\.jpg');
      
      // Allow create/update if:
      // 1. User is authenticated
      // 2. Filename matches pattern: userId_timestamp.jpg
      // 3. File size is reasonable (max 100KB for profile pics)
      // 4. Content type is image/jpeg
      allow write: if request.auth != null && 
                      fileName.matches(request.auth.uid + '_[0-9]+\\.jpg') &&
                      request.resource.size < 100 * 1024 && // 100KB max
                      request.resource.contentType == 'image/jpeg';
      
      // Allow read if user is authenticated
      // (Profile pictures should be viewable by authenticated users)
      // Alternative: allow read: if true; (for public access)
      allow read: if request.auth != null;
    }
    
    // Default deny for all other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
